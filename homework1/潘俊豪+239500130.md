# 作业一：
```cpp
#include <iostream>
using namespace std;

int main() {
    int num1, num2, choice;
    int num;
    // 输入两个数
    cout << "请输入第一个整数：";
    cin >> num1;
    cout << "请输入第二个整数：";
    cin >> num2;
    
    // 选择操作
    cout << "请选择操作（1:加法，2:减法）：";
    cin >> choice;
    
    // 根据选择执行相应的操作并输出结果
    if (choice == 1) {
        num=num1+num2;
       cout<<num1<<"+"<<num2<<"="<<num<<end1;
    } else if (choice == 2) {
        num=num1-num2;
        cout<<num1<<"-"<<num2<<"="<<num<<end1;
    } else {
        cout<<"error"<<end1;
    }
    
    return 0;
}
```
