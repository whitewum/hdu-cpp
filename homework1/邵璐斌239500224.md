using namespace std;
int main(){
	int num1,num2,choice;

    std::cout<<"请输入第一个整数:";
	std::cin>>num1;
	
	std::cout<<"请输入第二个整数:";
	std::cin>>num2;
	
	std::cout<< "请选择操作: " << std::endl;
	std::cout<<"1. 加法" << std::endl;
	std::cout<<"2. 减法" << std::endl;
	std::cin>>choice;
	if(choice==1){
		std::cout<<"结果："<<num1<<"+"<<num2<<"="<<(num1+num2)<<std::endl;
	}
	else if(choice==2){
		std::cout<<"结果："<<num1<<"-"<<num2<<"="<<(num1-num2)<<std::endl;
	}
	else{
		std::cout<<"无效的选择，请输入1或2。" <<std::endl; 
	}
	return 0;
}
